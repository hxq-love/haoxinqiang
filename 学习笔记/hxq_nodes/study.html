					添加CSS样式

					.fujiyemian{
							width: 100%;
								height: 400px;     --------设置父级为relative定位
								position: relative;
						}
						.fujiyemian .fujiyemian_left{   
							float: left;
							width: 30%;             --------左边子页面给宽度和浮动
							height: 400px;
							background-color: red
						}
						.fujiyemian .fujiyemian_right{
							float: left;
							width: 70%;				---------右边的子页面给宽度和浮动
							height: 400px;
							background-color: green;
						}
						.fujiyemian .qiehuan{      ------------脱离文档流的切换按钮
							background-color: white;
							position: absolute;--固定定位
							width: 50px;
							text-align: center;
							right: 70%;  ------初始位置
							height: 40px;
							line-height: 40px;
							top:180px;
							cursor: pointer;
						}
						.qiehuan_left{    -----点击切换后，按钮距离页面左边距离为0
							left:0px;
						}


						js代码：
						$(".fujiyemian .qiehuan").click(function(){  --0--按钮的点击事件
							var flg=$(".fujiyemian .qiehuan").attr("flag");   -----获取按钮自定义属性
							switch(flg)
							{
								case "0":  ---处于初始页面时
								$(this).addClass("qiehuan_left"); ---按钮靠页面左边缘
								$(this).siblings(".fujiyemian_left").css("display","none");---左子页面隐藏
								$(this).siblings(".fujiyemian_right").css("width","100%");---右子页面宽度100%；
								$(this).attr("flag","1"); --更换按钮状态
								break;

								case "1":  -----恢复页面
									
								$(this).removeClass("qiehuan_left"); ---恢复按钮原位置，距离左边30%
								$(".fujiyemian_left").css("display","block");--左子页面显示
								$(this).siblings(".fujiyemian_right").css("width","70%");--右子页面恢复原70%宽度
								$(this).attr("flag","0");  ---重新给予按钮状态
								
							}
	})
